SELECT resource_type, resource_associated_entity_id,request_status, request_mode,request_session_id,
    resource_description, o.object_id, o.name, o.type_desc ,DB_NAME(resource_database_id)
FROM sys.dm_tran_locks l, sys.objects o WHERE l.resource_associated_entity_id = o.object_id --and resource_database_id = DB_ID()
--and request_session_id in (4748)



SELECT DB_NAME(resource_database_id),request_session_id,count(*),'kill ' +convert(varchar(10),request_session_id)
FROM sys.dm_tran_locks l, sys.objects o WHERE l.resource_associated_entity_id = o.object_id --and resource_database_id = DB_ID()
group by DB_NAME(resource_database_id),request_session_id
