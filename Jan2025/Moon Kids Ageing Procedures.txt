CREATE  PROC [dbo].[udp_DesignAgegingUpdate]
as
begin
update muCore_Design set Aging=CONCAT(DATEDIFF(dd,dbo.fCore_IntToDate(InquiryDate),GETDATE()),' ','Days') 
where iMasterId>0
end
go
